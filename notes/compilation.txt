package.json
  "main" -  CJS or hybrid entrypoint
  "module": ESM entry point
  "browser": CJS entry point
  "types": for correct TypeScript type export in your package.json
  
  
tsconfig:
   "moduleResolution" - set to "node" for both CJS and ESM
   "allowSyntheticDefaultImports" - true for ESM (that can also work with CJS modules)
   "esModuleInterop" - true for CJS
